# service 01 ClusterIP
```
apiVersion: v1
kind: Pod
metadata:
  name: nginx
  namespace: ((네임스페이스 ))
  labels:
    app.kubernetes.io/name: proxy
spec:
  containers:
  - name: nginx
    image: nginx:stable
    ports:
      - containerPort: 80
        name: http-web-svc

---
apiVersion: v1
kind: Service
metadata:
  name: nginx-service
  namespace: ((네임스페이스 ))
spec:
  selector:
    app.kubernetes.io/name: proxy
  ports:
  - name: name-of-service-port
    protocol: TCP
    port: 80
    targetPort: http-web-svc
```

```
$ kubectl apply -f pod-service.yaml
```
